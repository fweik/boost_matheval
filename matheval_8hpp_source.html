39 /** @brief Sign function

42  * function.
43  *
44  * @f[
45  * \mathop{\mathrm{sgn}}(x) =
46  * \begin{cases}

179 /** @brief Evaluate the Abstract Syntax Tree
180  *
181  * This visits all the variants of the AST and applies the stored
182  * operators.
183  */
184 template < typename real_t >
185 class eval_ast
186 {
187 public:
188  /** @brief Necessary typedef for `boost::apply_visitor` */
189  using result_type = real_t;
190 
191  /** @brief Type of the symbol table */
192  using symbol_table_t = std::unordered_map<std::string, result_type>;
193 
194  /** @brief Constructor
195  *
196  * Saves the symbol table to apply variables.
197  */
198  eval_ast(symbol_table_t const &sym) : st(sym) {}

200  /** @brief Empty nodes in the tree evaluate to 0 (probably shouldn't happen) */
201 #if BOOST_VERSION >= 105600
202  result_type operator()(qi::info::nil_) const { return 0; }
203 #else
204  result_type operator()(qi::info::nil) const { return 0; }
205 #endif
206 
207  /** @brief Numbers evaluate to themselves */
208  result_type operator()(result_type n) const { return n; }
209 
210  /** @brief Variables evaluate to their value in the symbol table */
211  result_type operator()(std::string const &c) const
212  {
213  auto it = st.find(c);
214  if(it == st.end())
215  throw std::invalid_argument("Unknown variable " + c);
216  return it->second;
217  }
218 
219  /** @brief Recursively evaluate the AST */
220  result_type operator()(expr_ast<real_t> const& ast) const
221  {
222  return boost::apply_visitor(*this, ast.tree);
223  }

226  result_type operator()(binary_op<real_t> const& tree) const
227  {
228  return tree.op(

232  }

235  result_type operator()(unary_op<real_t> const& tree) const
236  {
237  return tree.op(

240  }

242 private:
243  symbol_table_t st;
244 };

369  grammar() : grammar::base_type(expression)
370  {

377  using boost::spirit::qi::_2;

379  using boost::spirit::qi::_val;

381  using boost::spirit::qi::alnum;

383 
384  boost::phoenix::function<unary_expr_<real_t>> unary_expr;
385  boost::phoenix::function<binary_expr_<real_t>> binary_expr;
386 
387  auto fmod = static_cast<real_t(*)(real_t,real_t)>(&std::fmod);
388  auto pow = static_cast<real_t(*)(real_t,real_t)>(&std::pow);
389 
390  expression =
391  term [_val = _1]
392  >> *( ('+' >> term [_val += _1])
393  | ('-' >> term [_val -= _1])
394  )
395  ;
396 
397  term =
398  factor [_val = _1]
399  >> *( ('*' >> factor [_val *= _1])
400  | ('/' >> factor [_val /= _1])
401  | ('%' >> factor [_val = binary_expr(fmod, _val, _1)])
402  )
403  ;
404 
405  factor =
406  primary [_val = _1]
407  >> *( ("**" >> factor [_val = binary_expr(pow, _val, _1)])
408  )
409  ;
410 
411  variable =
412  raw[lexeme[alpha >> *(alnum | '_')]];
413 
414  primary =
415  real [_val = _1]
416  | '(' >> expression [_val = _1] >> ')'
417  | ('-' >> primary [_val = unary_expr(std::negate<real_t>{}, _1)])
418  | ('+' >> primary [_val = _1])
419  | (ufunc >> '(' >> expression >> ')')
420  [_val = unary_expr(_1, _2)]
421  | (bfunc >> '(' >> expression >> ',' >> expression >> ')')
422  [_val = binary_expr(_1, _2, _3)]
423  | constant [_val = _1]
424  | variable [_val = _1]

432 /** @brief Class interface
433  *
434  * This class hides the grammar, AST, and AST traversal behind some
435  * member functions.
436  *
437  * @tparam real_t datatype of the result
438  */
439 template < typename real_t >
440 class Parser
441 {
442  detail::expr_ast<real_t> ast;
443 public:
444  /** @brief Parse an expression
445  *
446  * This function builds the grammar and parses the iterator into
447  * an AST.
448  *
449  * @param[in] first iterator to the start of the input sequence
450  * @param[in] last iterator to the end of the input sequence
451  */
452  template < typename Iterator >
453  void parse(Iterator first, Iterator last)
454  {
455  static detail::grammar<real_t,Iterator> const g;
456 
457  ast = detail::expr_ast<real_t>{}; // Drop old AST
458 
459  bool r = boost::spirit::qi::phrase_parse(
460  first, last, g,
461  boost::spirit::ascii::space, ast);
462 
463  if (!r || first != last)
464  {
465  std::string rest(first, last);
466  throw std::runtime_error("Parsing failed at " + rest);
467  }
468  }
469 
470  /** @overload parse(Iterator first, Iterator last) */
471  void parse(std::string const &str)
472  {
473  parse(str.begin(), str.end());
474  }
475 
476  /** @brief Evaluate the AST with a given symbol table
477  *
478  * @param[in] st the symbol table for variables
479  */
480  real_t evaluate(typename detail::eval_ast<real_t>::symbol_table_t const &st)
481  {
482  detail::eval_ast<real_t> solver(st);
483  return solver(ast);
484  }
485 };
